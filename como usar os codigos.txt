# Para ver se é um repositório
$ git status

# Inicializar um repositório
$ git init

# Adicionar origem remota (github)
$ git add remote origin https://github.com.....

# Adicionar (rastrear) arquivos no repositório
$ git add .

# Confirmar alterações
$ git commit -a -m "sua mensagem aqui"

# Enviar ao github
$ git push origin master

# Clonar um repositório do github
$ git clone https://github.com.....
